cmake_minimum_required(VERSION 3.16)
project(Oculox_DrowsinessDetector_QtOnly LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_AUTOUIC ON)

# Find Qt6 with better error handling
find_package(Qt6 COMPONENTS Widgets QUIET)
if(NOT Qt6_FOUND)
    message(STATUS "Qt6 not found. Trying common installation paths...")
    set(CMAKE_PREFIX_PATH 
        "C:/Qt/6.9.3/msvc2022_64"
        "C:/Qt/6.8.0/msvc2022_64" 
        "C:/Qt/6.7.0/msvc2022_64"
        ${CMAKE_PREFIX_PATH})
    find_package(Qt6 COMPONENTS Widgets REQUIRED)
endif()

# Compiler warnings settings
if(MSVC)
    add_compile_options(/W4 /WX-)  # High warning level, warnings not as errors
    add_compile_definitions(_CRT_SECURE_NO_WARNINGS)
else()
    add_compile_options(-Wall -Wextra -Wpedantic)
endif()

# Mock OpenCV types for Qt-only compilation test
add_compile_definitions(OPENCV_MOCK_BUILD)

add_executable(Oculox_DrowsinessDetector_QtOnly
    main.cpp
    src/oculox_window.cpp
    src/oculox_window.h
)

target_link_libraries(Oculox_DrowsinessDetector_QtOnly
    Qt6::Widgets
)
